// Apply persisted theme synchronously before first paint to avoid a // light-mode flash for users whose preference is dark (which is the // default for new accounts). Reads `volant.theme` from localStorage // and falls back to "dark". (function () { try { var t = localStorage.getItem("volant.theme"); if (t !== "light") { document.documentElement.classList.add("dark"); } } catch (e) { document.documentElement.classList.add("dark"); } })();
